void sumMatrix(double** matrix, int size1, int size2, double* arr) {
    for (int i = 0; i < size1; i++) {
        for (int j = 0; j < size2; j++) {
            if ((i < size1 / 2 && j < size2 / 2) || (i > size1 / 2 && j > size2 / 2))
            {
                arr[j] = arr[j] + *(*(matrix + i) + j);
            }
        }
    }

}
